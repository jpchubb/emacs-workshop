\section{Key}
\begin{tabular}{@{}ll@{}}
  $<$B$>$ & Backspace.\\
  $<$S$>$ & Space.\\
  $<$T$>$ & Tab.\\
  $<$E$>$ & Escape.\\
  $<$n$>$ & Number.\\
  $<$c$>$ & Command.\\
\end{tabular}

%% \section{GNU Emacs}
\subsection{Cursor Movement}
\begin{tabular}{@{}ll@{}}
  {C-l} & Display current point at center of screen.\\
  {C-f} & Move forward a character.\\
  {C-b} & Move backwards a character.\\
  {M-f} & Move forwards a word.\\
  {M-b} & Move backwards a word.\\
  {C-n} & Move to next line.\\
  {C-p} & Move to previous line.\\
  {C-a} & Move to beginning of line.\\
  {C-e} & Move to end of line.\\
  {M-a} & Move back to beginning of sentence.\\
  {M-e} & Move forward to end of sentence.\\
  {M-$<$} & Move to beginning of buffer.\\
  {M-$>$} & Move to end of buffer.\\
\end{tabular}
\subsection{Windows, Buffers and Files}
\begin{tabular}{@{}ll@{}}
  {C-x 0} & Kill current window.\\
  {C-x 1} & Kills all windows except for the current one.\\
  {C-x 2} & Splits the window horizontal.\\
  {C-x 3} & Splits the window vertical.\\
  {C-x o} & Move to next window.\\
  {C-M-v} & Scroll next window down.\\
  {C-x s} & Save buffers.\\
  {C-x C-w} & Write buffer to name.\\
  {C-x C-v} & Find alternate file.\\
  {C-x i} & Insert file at cursor.\\
  {C-x b} & Switch to buffer.\\
  {C-x k} & Kill current buffer.\\
  {C-v} & Scroll buffer down.\\
  {M-v} & Scroll buffer up.\\
  {C-x C-b} & List buffers.\\
  {C-x C-f} & Find file.\\
  {C-x C-s} & Save file.\\ 
\end{tabular}
\subsection{External Commands}
\begin{tabular}{@{}ll@{}}
  {M-!} & Execute shell command.\\
  {M-$|$} & Execute shell command on region.\\
  {M-x compile} & Run \texttt{make} over buffer.\\
  {M-x shell} & Start a shell session.\\
  {M-x gdb} & Start \texttt{gdb} in a new buffer.\\
  {M-x grep} & Invokes \texttt{grep} and prints results.\\
  {M-x man} & View \texttt{man} page.\\
\end{tabular}
\subsection{Keyboard Macros}
\begin{tabular}{@{}ll@{}}
  {C-x (} & Start recording macro.\\
  {C-x )} & Finish recording macro.\\
  {C-x e} & Call last set keyboard macro.\\
\end{tabular}
\subsection{Manipulate Text}
\begin{tabular}{@{}ll@{}}
  {C-d} & Delete the next character after the cursor.\\
  {M-$<$B$>$} & Kill the word previous to the cursor.\\
  {M-d} & Kill the next word after the cursor.\\
  {C-k} & Kill to end of line.\\
  {C-u 0 C-k} & Kill to beginning of line.\\
  {M-k} & Kill to the end of the current sentence.\\
  {C-w} & Move the selected text to a buffer.\\
  {M-w} & Copy the selected text to a buffer.\\
  {C-y} & Paste the contents of the buffer to display.\\
  {M-z} & Zap until character.\\
  {C-t} & Transpose two characters.\\
  {M-t} & Transpose two words.\\
  {C-x C-t} & Transpose two lines.\\
  {M-c} & Uppercase words from cursor position.\\
  {M-l} & Lowercase words from cursor position.\\
  {M-c} & Uppercase character at cursor position.\\
\end{tabular}
\subsection{General Commands}
\begin{tabular}{@{}ll@{}}
  {C-x u} & Undo.\\
  {C-g} & Kill current command.\\
  {C-x C-c} & Close GNU Emacs.\\
  {C-z} & Suspend GNU Emacs.\\
  {C-s} & Search forwards.\\
  {C-r} & Search backwards.\\
  {C-s C-w} & Search for word under cursor.\\
  {M-\%} & Search and replace.\\
  {C-M-\%} & Search and replace (regex).\\ 
  {C-u $<$n$>$ $<$c$>$} & Repeat $<$c$>$ $<$n$>$ times.\\
  {M-/} & Auto-complete.\\
  {C-x z} & Repeat previous command.\\
  {M-;} & Comment selected region.\\
  {C-$<$S$>$} & Enter select mode.\\
  {M-@} & Select word.\\
  {M-h} & Select paragraph.\\
  {M-g} & Go to line.\\
  {C-x f} & Set column fill limit.\\
  {M-q} & Fill paragraph.\\
  {M-\$} & Check spelling of word.\\
\end{tabular}
\subsection{Modal Commands}
\subsubsection{Major Modes}
\begin{tabular}{@{}ll@{}}
  {M-x latex-mode} & LaTeX mode.\\
  {M-x haskell-mode} & Haskell mode.\\
  {M-x c-mode} & C mode.\\
  {M-x python-mode} & Python mode.\\
  {M-x text-mode} & Text mode.\\
  {M-x lisp-mode} & Lisp mode.\\
\end{tabular}
\subsubsection{Minor Modes}
\begin{tabular}{@{}ll@{}}
  {M-x flyspell-mode} & Automatic spell check.\\
  {M-x flymake-mode} & On-the-fly syntax checker.\\
\end{tabular}
\subsubsection{{\LaTeX} Mode}
\begin{tabular}{@{}ll@{}}
  {C-c C-e} & Close {\LaTeX} block.\\
  {C-c C-o} & Insert {\LaTeX} block.\\
  {C-c C-f} & Compile {\LaTeX} to PDF.\\
  {C-j} & End paragraph.\\
\end{tabular}
\subsubsection{Haskell Mode}
\begin{tabular}{@{}ll@{}}
  {C-c C-l} & Start GHCi session.\\
\end{tabular}
\subsubsection{C Mode}
\begin{tabular}{@{}ll@{}}
  {C-c C-a} & Electrification.\\
  {M-a} & Beginning of statement.\\
  {M-e} & End of statement.\\
  {M-C-a} & Beginning of function.\\
  {M-C-e} & End of function.\\
  {C-j} & Insert a new line and indent.\\
\end{tabular}
\subsubsection{Python Mode}
\begin{tabular}{@{}ll@{}}
  {C-c !} & Start an interactive Python session.\\
  {C-j} & Insert a new line and indent.\\
  {C-M-a} & Go to the beginning of the function or class.\\
  {C-M-e} & Go to the end of the function or class.\\
  {C-M-h} & Select the function or class.\\
\end{tabular}
\subsubsection{Text Mode}
\begin{tabular}{@{}ll@{}}
  {M-s} & Center line.\\
  {M-S} & Center paragraph.\\
\end{tabular}
\subsubsection{Lisp Mode}
\begin{tabular}{@{}ll@{}}
  {C-x C-e} & Evaluate expression.\\
\end{tabular}
\subsection{Help Commands}
\begin{tabular}{@{}ll@{}}
  {C-h a} & Apropos help.\\
  {C-h b} & Show key bindings.\\
  {C-h c} & Show command key-stroke invokes.\\
  {C-h f} & Describe function.\\
  {C-h F} & Show GNU Emacs FAQ.\\
  {C-h i} & Info help.\\
  {C-h k} & Describe key.\\
  {C-h m} & Describe mode.\\
  {C-h p} & List GNU Emacs packages.\\
  {C-h t} & Run tutorial.\\
  {C-h v} & Show variable information.\\
  {C-h w} & Show key-stroke command.\\
  {C-h ?} & General help.\\
\end{tabular}
